% Placeholder for missing personal info — visibly flagged, easy to grep
\newcommand{\TODO}[1]{\textbf{\color{red}[CHƯA CÓ — #1]}}

% Figure include helper (centered, width-controlled)
\newcommand{\fig}[3]{% {file}{width}{caption+label inside}
  \begin{figure}[H]\centering
  \includegraphics[width=#2\linewidth]{#1}
  #3
  \end{figure}}

% Screenshot helper: show the image if present, else a labelled placeholder box.
% Usage: \screenshot{figures/screenshots/home.png}{0.85}{Giao diện trang chủ}{fig:home}
\newcommand{\screenshot}[4]{%
  \begin{figure}[H]\centering
  \IfFileExists{#1}
    {\fbox{\includegraphics[width=#2\linewidth,height=0.62\textheight,keepaspectratio]{#1}}}
    {\fbox{\parbox[c][5cm][c]{#2\linewidth}{\centering \TODO{ảnh: #1}}}}%
  \caption{#3}\label{#4}
  \end{figure}}

% --- Use-case specification table ---------------------------------
% Two-column "field : value" rows that may span pages (longtable).
\newenvironment{ucspec}[1]% #1 = caption text
{%
  \par\vspace{4pt}
  \footnotesize
  \setlength{\extrarowheight}{2pt}
  \def\uccaption{#1}%
  \begin{longtable}{|>{\raggedright\arraybackslash}p{0.26\linewidth}|>{\raggedright\arraybackslash}p{0.66\linewidth}|}
  \hline
}
{%
  \hline
  \caption{\uccaption}\\
  \end{longtable}
  \normalsize
}
% A simple field/value row
\newcommand{\ucrow}[2]{\textbf{#1} & #2 \\ \hline}

% --- DB design table (STT | Tên trường | Kiểu | Ràng buộc | Extra) ---
\newenvironment{dbtable}[1]% #1 = caption text
{%
  \par\vspace{4pt}\footnotesize
  \setlength{\extrarowheight}{2pt}
  \def\dbcaption{#1}%
  \begin{longtable}{|c|>{\raggedright\arraybackslash}p{0.22\linewidth}|>{\raggedright\arraybackslash}p{0.20\linewidth}|>{\raggedright\arraybackslash}p{0.16\linewidth}|>{\raggedright\arraybackslash}p{0.24\linewidth}|}
  \hline
  \rowcolor{gray!12}\textbf{STT} & \textbf{Tên trường} & \textbf{Kiểu dữ liệu} & \textbf{Ràng buộc} & \textbf{Extra} \\ \hline
  \endhead
}
{%
  \caption{\dbcaption}\\
  \end{longtable}
  \normalsize
}
